# 1.分解质因数。

```js
function f1(num) {
  var num1 = num;
  var result = [];
  var i = 1;

  while (i < num1) {
    i += 1;
    while (num1 % i == 0) {
      num1 /= i;
      result.push(i);
    }
  }

  return result;
}
```
# 2.斐波那契数列第n项，第一项视为1。
```js
// 递归版
function fibonacci(n){
  if(n <=1 ){
    return n;
  }
  return fibonacci(n - 1) + fibonacci(n - 2);
}
// 非递归版
function fibonacci(n){
  if(n <= 1){
    return n;
  }
  var f = 1;
  var g = 0;
  while(n >= 1){
    f = f + g;
    g = f - g;
    n--;
  }
  return g;
}
```
